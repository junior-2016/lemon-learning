513 static int handleflags(int i,FILE*err) {
514  int v;

516  int j;
517  for (j = 0; op[j].label; j++) {
518  // 遍历lemon支持的选项数组,若找出一个与命令行参数的值匹配的就立即退出.
519  // 因此for循环退出后,若op[j].label不为0,则代表匹配成功.
520  if (strncmp(&argv[i][1], op[j].label, lemonStrlen(op[j].label)) == 0) break;
521  }
522  v=((argv[i][0]=='-')?1:0); // 若选项带"-"前缀则v为1,若前缀为"+"则v为0
523  if (op[j].label==0){// 匹配不成功
524  if (err){
525  fprintf(err,"%sundefined option.\n",emsg);
526  errline(i,1,err);

528  errcnt++;

530  // 如果选项没有相应的附加参数地址,忽略该选项
531  }else if(op[j].type==OPT_FLAG){
532  // 如果选项类型是OPT_FLAG,就把v的值写入这个选项的附加参数地址所在的地方.
533  // 同时选项的附加参数arg此时被强制转换为int*,说明此时arg指针指向的是一个int值
534  *((int*)op[j].arg) = v;
535  }else if(op[j].type==OPT_FFLAG){
536  // 如果选项类型是OPT_FFLAG,首先把选项附加的参数指针arg强制转换为
537  // (void(*)(int))类型的函数指针,然后把v的值作为arg指向的函数的实参传入,
538  // 相当于直接调用arg指针指向的函数.所以这一步是一个函数调用过程而不是一个存值的过程.
539  // 注意后面代码的阅读顺序应该是: (*((void(*)(int))(op[j].arg)))(v);
540  (*(void(*)(int))(op[j].arg))(v);
541  }else if(op[j].type==OPT_FSTR){
542  // 如果选项类型是OPT_FSTR,首先把选项附加的参数指针arg强制转换为
543  // (void(*)(char*))类型的函数指针,然后传递&argv[i][2]调用这个函数指针指向的函数.
544  (*(void(*)(char*))(op[j].arg))(&argv[i][2]);
545  }else{
546  if (err){
547  fprintf(err,"%smissing argument on switch.\n",emsg);
548  errline(i,1,err);
549  }
550  errcnt++;
551  }

916 struct symbol *Symbol_find(const char *key){
917  unsigned h; // 用来储存哈希值
918  x2node *np; // symbol节点指针
919  if (x2a==0) return 0; // 这一步基本不会发生,因为前面已经通过Symbol_init()初始化了
920 
921  /*
922  * 因为x2a->size-1==127,这一步相当于 hash(x)&(1111111),这样就把hash()的范围约束在[0,127]里,
923  * 当然把大范围的hash()压缩为小范围必然出现哈希冲突,这个程序的排解方法是链表尾插排解冲突
924  */
925  h = strhash(key) & (x2a->size-1);
926 
927  np=x2a->ht[h]; // 在哈希表访问key对应的符号指针
928 
929  while (np){ // 链表尾插排解冲突
930  if (strcmp(np->key,key)==0) break; // 发现key相同的,直接返回对应的符号指针,
931  np=np->next; // 否则,继续链表的下一个指针,直到访问到空指针,说明之前不存在这个符号,就可以把新符号尾插到链表里
932  }
933  return np?np->data:0; // 注意返回的是符号指针而不是符号节点,所以要使用np->data获得节点储存的符号指针
934 }
